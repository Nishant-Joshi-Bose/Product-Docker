cmake_minimum_required(VERSION 2.8)

project(Eddie)

if(NOT DEFINED EDDIE_SOURCE_DIR)
        SET(EDDIE_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/)
endif(NOT DEFINED EDDIE_SOURCE_DIR)

if(NOT DEFINED OUTPUT_BIN_DIR)
        SET(OUTPUT_BIN_DIR ${EDDIE_SOURCE_DIR}/output/${CFG}/${CORE_TARGET}/bin)
        SET(CMAKE_INSTALL_PREFIX ${EDDIE_SOURCE_DIR})
endif(NOT DEFINED OUTPUT_BIN_DIR)

if(NOT DEFINED OUTPUT_LIB_DIR)
        SET(OUTPUT_LIB_DIR ${EDDIE_SOURCE_DIR}/output/${CFG}/${CORE_TARGET}/lib)
        SET(CMAKE_INSTALL_PREFIX ${EDDIE_SOURCE_DIR})
endif(NOT DEFINED OUTPUT_LIB_DIR)

SET(SOUNDTOUCH_SDK_NATIVE ${EDDIE_SOURCE_DIR}/SoundTouch-SDK-native)
SET(SOUNDTOUCH_SDK_TI ${EDDIE_SOURCE_DIR}/SoundTouch-SDK)
SET(HSP_DRIVERS ${EDDIE_SOURCE_DIR}/sm2-HSP/DeviceDrivers)
SET(HSP_INCLUDES ${EDDIE_SOURCE_DIR}/sm2-HSP/includes)

SET(BUILD_CFG_DIR ${EDDIE_SOURCE_DIR}/BuildConfig)
INCLUDE( exe.cmake )

MESSAGE("SOUNDTOUCH_SDK_NATIVE is ${SOUNDTOUCH_SDK_NATIVE}")
MESSAGE("SOUNDTOUCH_SDK_TI is ${SOUNDTOUCH_SDK_TI}")
MESSAGE("HSP_DRIVERS is ${HSP_DRIVERS}")
MESSAGE("OUTPUT_BIN_DIR is ${OUTPUT_BIN_DIR}")
